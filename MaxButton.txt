    private double calc(double x, String input, char opt) {
        inText.setFont(inText.getFont().deriveFont(Font.BOLD));
        double y = Double.parseDouble(input);
        if (opt == '+') {
            return x + y;
        } else if (opt == '-') {
            return x - y;
        } else if (opt == '*') {
            return x * y;
        } else if (opt == '/') {
            return x / y;
        } else if (opt == '%') {
            return x % y;
        }
        else if(opt == '^'){
        	return Math.pow(x,y);
        }
        else if(opt == 'G'){
        	return Math.max(x,y);
        }
        inText.setFont(inText.getFont().deriveFont(Font.PLAIN));
        return y;
    }
